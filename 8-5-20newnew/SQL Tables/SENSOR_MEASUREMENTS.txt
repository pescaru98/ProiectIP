CREATE TABLE `SENSOR_MEASUREMENTS` (
  `SENSOR_MEASUREMENT_ID` int(11) NOT NULL AUTO_INCREMENT,
  `USERNAME` varchar(50) NOT NULL,
  `MEASURE_DATE` date NOT NULL,
  `TEMPERATURE` double NOT NULL,
  `PULSE` int(11) NOT NULL,
  `HUMIDITY` double NOT NULL,
  `EKG` varchar(10000) NOT NULL,
  PRIMARY KEY (`SENSOR_MEASUREMENT_ID`),
  KEY `SENSOR_MEASUREMENTS_FK1_idx` (`USERNAME`),
  CONSTRAINT `SENSOR_MEASUREMENTS_FK1` FOREIGN KEY (`USERNAME`) REFERENCES `USERS` (`USERNAME`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=34 DEFAULT CHARSET=latin1;
