CREATE TABLE `USERS` (
  `USERNAME` varchar(50) NOT NULL,
  `PASSWORD` varchar(50) NOT NULL,
  `ACTIVE_ID` int(11) NOT NULL,
  `ROLE_ID` int(11) NOT NULL,
  PRIMARY KEY (`USERNAME`),
  KEY `USERS_FK1_idx` (`ROLE_ID`),
  KEY `USERS_FK2_idx` (`ACTIVE_ID`),
  CONSTRAINT `USERS_FK1` FOREIGN KEY (`ROLE_ID`) REFERENCES `T_ROLES` (`ROLE_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `USERS_FK2` FOREIGN KEY (`ACTIVE_ID`) REFERENCES `T_ACTIVES` (`ACTIVE_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
