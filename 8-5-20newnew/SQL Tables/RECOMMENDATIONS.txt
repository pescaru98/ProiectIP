CREATE TABLE `RECOMMENDATIONS` (
  `RECOMMENDATION_ID` int(11) NOT NULL AUTO_INCREMENT,
  `USERNAME` varchar(50) NOT NULL,
  `START_DATE` date NOT NULL,
  `END_DATE` date NOT NULL,
  `RECOMMEND_ID` int(11) NOT NULL,
  `COMMENTS` varchar(100) NOT NULL,
  PRIMARY KEY (`RECOMMENDATION_ID`),
  KEY `RECOMMENDATIONS_FK1_idx` (`USERNAME`),
  KEY `RECOMMENDATIONS_FK2_idx` (`RECOMMEND_ID`),
  CONSTRAINT `RECOMMENDATIONS_FK1` FOREIGN KEY (`USERNAME`) REFERENCES `USERS` (`USERNAME`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `RECOMMENDATIONS_FK2` FOREIGN KEY (`RECOMMEND_ID`) REFERENCES `T_RECOMMENDATION` (`RECOMMENDATION_ID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=latin1;
