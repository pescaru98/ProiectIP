CREATE TABLE `MAP_PACIENT_TO_DOC` (
  `MAP_ID` int(11) NOT NULL AUTO_INCREMENT,
  `PACIENT_ID` varchar(50) NOT NULL,
  `DOCTOR_ID` varchar(50) NOT NULL,
  PRIMARY KEY (`MAP_ID`),
  UNIQUE KEY `PACIENT_ID_UNIQUE` (`PACIENT_ID`,`DOCTOR_ID`),
  KEY `DOCTOR_FK1_idx` (`DOCTOR_ID`),
  CONSTRAINT `DOCTOR_FK1` FOREIGN KEY (`DOCTOR_ID`) REFERENCES `DOCTORS` (`USERNAME`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `PACIENT_FK1` FOREIGN KEY (`PACIENT_ID`) REFERENCES `PACIENTS` (`USERNAME`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=latin1;
