CREATE TABLE `MANUAL_MEASUREMENTS` (
  `MANUAL_MEASUREMENT_ID` int(11) NOT NULL AUTO_INCREMENT,
  `USERNAME` varchar(50) NOT NULL,
  `MEASURE_DATE` date NOT NULL,
  `TEMPERATURE` double NOT NULL,
  `WEIGHT` double NOT NULL,
  PRIMARY KEY (`MANUAL_MEASUREMENT_ID`),
  KEY `MANUAL_MEASUREMENTS_FK1_idx` (`USERNAME`),
  CONSTRAINT `MANUAL_MEASUREMENTS_FK1` FOREIGN KEY (`USERNAME`) REFERENCES `USERS` (`USERNAME`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=latin1;
