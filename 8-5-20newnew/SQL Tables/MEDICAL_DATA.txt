CREATE TABLE `MEDICAL_DATA` (
  `MEDICAL_DATA_ID` int(11) NOT NULL AUTO_INCREMENT,
  `USERNAME` varchar(50) NOT NULL,
  `MEDICAL_HISTORY` varchar(10000) NOT NULL,
  `ALLERGY` varchar(10000) NOT NULL,
  `CARDIO_CONSULTATION` varchar(10000) NOT NULL,
  PRIMARY KEY (`MEDICAL_DATA_ID`),
  KEY `MEDICAL_DATA_FK1_idx` (`USERNAME`),
  CONSTRAINT `MEDICAL_DATA_FK1` FOREIGN KEY (`USERNAME`) REFERENCES `USERS` (`USERNAME`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=latin1;
